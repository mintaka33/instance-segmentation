## set python virtual env
```bash
sudo apt update && sudo apt upgrade
sudo apt install python3-pip

sudo pip3 install virtualenv

mkdir $(workdir)
cd $(workdir)

virtualenv --no-site-packages envmask
source envmask/bin/activate

```
